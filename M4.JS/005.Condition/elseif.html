lệnh if else if else trong js

khi nào sử dụng else if : khi mà có nhiều hơn 1 điều kiện

    cú pháp lện if else if else

    if (điều kiện 1){
        // điều kiện 1 == true thì code trong {} sẽ chạy
    } else if (điều kiện 2){
        // điều kiên 2 == true thì code trong {} sẽ chạy
    } else if (điều kiện 3){
        // điều kiên 3 == true thì code trong {} sẽ chạy
    } else if (điều kiện 4){
        // điều kiên 4 == true thì code trong {} sẽ chạy
    }else {
        // các trường hợp còn lại nếu tất cả các điều kiện trên đều sai thì cde {} sẽ chạy
    }
